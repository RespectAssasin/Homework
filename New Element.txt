private void Button_Click(object sender, RoutedEventArgs e)
{
    CreateNewRecordElement();
}

private void CreateNewRecordElement()
{
    Grid newGrid = new Grid();
    newGrid.Width = BEST_Window.Width;



    for (int i = 0; i < _numberOfColums; i++)
    {
        ColumnDefinition column = new ColumnDefinition();
        column.Width = new GridLength(newGrid.Width / _numberOfColums);
        newGrid.ColumnDefinitions.Add(column);
    }
    
    for (int i = 0; i < _numberOfColums; i++)
    {
        TextBlock newTextBlock = new TextBlock();

        newGrid.Children.Add(newTextBlock);
        newTextBlock.Text = "LMAO";

        Grid.SetColumn(newTextBlock, i);
    }

    ButtonsPanel.Children.Add(newGrid);
}
